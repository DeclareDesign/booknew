# first a designer that returns a single study as a function of the global mu (effect size) parameter and the sd of effect sizes (tau)
# incorporates measurement error across sites (uncorrelated with effect sizes)
study_designer <- function(effect_size, effect_size_sd) {
  effect_size <- rnorm(1, mean = effect_size, sd = effect_size_sd)
  design <-
    declare_population(N = 100, 
                       latent_outcome = rnorm(N),
                       measurement_noise = rnorm(N)) +
    declare_potential_outcomes(Y ~ effect_size * Z + latent_outcome + measurement_noise) +
    declare_potential_outcomes(Y_latent ~ Z + latent_outcome) +
    declare_estimand(ATE = effect_size) +
    declare_assignment(prob = 0.5) +
    declare_reveal(Y, Z) +
    declare_estimator(Y ~ Z, model = difference_in_means, estimand = "ATE")
}

# function that runs the meta studies together
meta_study_estimates_data <- function(effect_size, effect_size_sd) {
  study_designs <- expand_design(study_designer, effect_size = effect_size, effect_size_sd = effect_size_sd)
  simulate_design(study_designs, sims = 1)
}

rma_tidy <- function(data){
  # function to return the mu and tau^2 estimates
  fit <- rma(yi = estimate, sei = std.error, data = data, method = "REML")
  mu <- with(summary(fit), data.frame(estimator_label = "mu", estimand_label = "mu", estimate = beta, std.error = se, statistic = zval, p.value = pval, conf.low = ci.lb, conf.high = ci.ub))
  tau <- with(summary(fit), data.frame(estimator_label = "tau_sq", estimand_label = "tau_sq", estimate = tau2, std.error = se.tau2))
  bind_rows(mu, tau)
}

meta_study_design <- 
  declare_population(effect_size = rep(.25, 50), effect_size_sd = .1, handler = meta_study_estimates_data) +
  declare_estimand(mu = .25, tau_sq = .1^2) +
  declare_estimator(handler = rma_tidy)
